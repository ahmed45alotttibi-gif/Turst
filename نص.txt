function updateCartUI() {
  const list = document.getElementById('cart-items');
  list.innerHTML = "";
  cart.forEach((item, i) => {
    let li = document.createElement('li');
    li.style.padding = "10px 0";
    li.style.borderBottom = "1px solid #222";
    li.innerText = (i + 1) + ". " + item;
    list.appendChild(li);
  });
}

}
